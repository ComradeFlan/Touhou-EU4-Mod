	}

		}
	}

		}

#Beating Tenshi up
country_event = {
	id = flavor_hkr.3
	title = flavor_hkr.3.t
	desc = flavor_hkr.3.desc
	picture = LIBERUM_VETO_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_hkr.3.a
		custom_tooltip = flavor_hkr.3.a.tt
		hidden_effect = {
			every_known_country = {
				limit = {
					religion = ROOT
					NOT = { tag = HVR }
					NOT = { alliance_with = HVR }
				}
				add_opinion = {
					who = ROOT
					modifier = th_beat_tenshi_up
				}
			}
		}
		
		ai_chance = { 
			factor = 100
		}
	}
}

#Immaterial and Missing Power(currently only one flavor event. We may extend it to a minor incident in the future)

	title = flavor_hkr.4.t
	desc = flavor_hkr.4.desc

		is_year = 1493

		}
	}

		}

			female = yes
		}

			set_country_flag = hkr_suika_general       #Has chosen Suika to be general

						id = flavor_hkr.400  #Suika returned and available again

					}
				}
				25 = {

						id = flavor_hkr.400  #Suika returned and available again
						days = 14600
					}
				}
				25 = {
					country_event = {
						id = flavor_hkr.400  #Suika returned and available again
						days = 18250
					}
				}
				25 = {
					country_event = {
						id = flavor_hkr.400  #Suika returned and available again
						days = 21900
					}
				}
			}
		}
		
		ai_chance = { 
			factor = 50
		}
	}

		name = flavor_hkr.4.b

		}

			factor = 50
		}
	}
