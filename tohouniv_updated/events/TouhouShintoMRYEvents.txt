#Activation event
country_event = {
	id = TH_mry_shinto_events.2
	title = TH_mry_shinto_events.2.t
	desc = TH_mry_shinto_events.2.desc
	picture = HAKUREI_eventPicture

	hidden = yes

	trigger = {
		owns = 4803
		religion = Moriya_Shintoism
		4803 = {
			NOT = { has_province_flag = th_mry_shinto_is_buffing_deity }
			religion = Moriya_Shintoism
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {	#Activates modifier for every Moriya Shinto nation
		name = TH_mry_shinto_events.2.a
		4803 = {
			set_province_flag = th_mry_shinto_is_buffing_deity
		}
		every_country = {
			limit = {
				religion = Moriya_Shintoism
			}
			country_event = {
				id = TH_mry_shinto_events.3
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}
